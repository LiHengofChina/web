const StatisticsSummary: React.FC = () => {
	
  // 🔽 这里是组件内部，调用 Hook 获取数据
	const { data, loading } = useRequest(
	  () => summaryStatistics({ timeRange }),
	  {
		formatResult: (res) => res.data || [],
		refreshDeps: [timeRange],
	  },
	);
  return <div>...</div>
}

//================================================
//================================================

useRequest 是个用于数据获取的 hook，

这是在一个 React 函数组件中，使用 useRequest 获取异步数据的写法。

在函数组件中调用了 useRequest 这个 React Hook。
