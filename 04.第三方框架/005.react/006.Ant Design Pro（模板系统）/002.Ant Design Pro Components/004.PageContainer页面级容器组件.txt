


PageContainer 是 Ant Design Pro 提供的 页面级容器组件，常用于布局页面的结构和标题区域。


import { PageContainer } from '@ant-design/pro-components';

const MyPage = () => {
  return (
    <PageContainer
      header={{
        title: '用户管理',
        breadcrumb: {},
      }}
    >
      <div>页面内容</div>
    </PageContainer>
  );
};
